# method 1: queue BFS
- use 2d array indicate whether O is connected to border, use BFS traverse from border O

## complexity analysis
- time: O(mn)
- space: O(mn)

## code
```
class Solution {
public:
    void solve(vector<vector<char>>& board) {
        if (board.empty()) return;
        int height = board.size();
        int width = board[0].size();
        if (height == 0 || width == 0) return;
        vector<vector<bool>> table(height, vector<bool>(width, false));
        queue<pair<int,int>> q;
        
        // set border O to true
        for (int i = 0; i < height; i++) {
            if (board[i][0] == 'O') {
                table[i][0] = true;
                q.push(make_pair(i,0));
            }
            if (board[i][width - 1] == 'O') {
                table[i][width - 1] = true;
                q.push(make_pair(i, width - 1));
            }
        }
        for (int j = 0; j < width; j++) {
            if (board[0][j] == 'O') {
                table[0][j] = true;
                q.push(make_pair(0, j));
            } 
            if (board[height - 1][j] == 'O') {
                table[height - 1][j] = true;
                q.push(make_pair(height - 1, j));
            }
        }
        
        while (!q.empty()) {
            pair<int,int> pos = q.front();
            q.pop();
            for (int i = 0; i < 4; i++) {
                if (pos.first+x[i] < 0 || pos.first+x[i] >= height || pos.second+y[i] < 0 || pos.second+y[i] >= width) continue;
                if (board[pos.first+x[i]][pos.second+y[i]] == 'O' && table[pos.first+x[i]][pos.second+y[i]] == false) {
                    table[pos.first+x[i]][pos.second+y[i]] = true;
                    q.push(make_pair(pos.first+x[i], pos.second+y[i]));
                }
            }
        }
        
        for (int i = 0; i < height; i++) {
            for (int j = 0; j < width; j++) {
                if (table[i][j] == false && board[i][j] == 'O') {
                    board[i][j] = 'X';
                }
            }
        }
        return;
        
        
    }
private:
    vector<int> x{-1,0,1,0};
    vector<int> y{0,-1,0,1};
};
```
