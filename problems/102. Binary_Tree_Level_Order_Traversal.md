# method: queue
- use queue to level order traversal, size of q for each iteration is level size

## complexity analysis
- time: O(n)
- space: O(n)

## code
```
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;
        queue<TreeNode*> q;
        q.push(root);
        int size;
        TreeNode* cur;
        while (!q.empty()) {
            size = q.size();
            vector<int> v;
            while (size > 0) {
                cur = q.front();
                v.push_back(cur->val);
                q.pop();
                if (cur->left != nullptr) q.push(cur->left);
                if (cur->right != nullptr) q.push(cur->right);
                size--;
            }
            res.push_back(v);
        }
        return res;
    }
};
```
